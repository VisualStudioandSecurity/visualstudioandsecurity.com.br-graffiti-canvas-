    <script>
        const box = document.getElementById('game-box');

        function spawn() {
            const t = document.createElement('div');
            t.className = 'threat';
            t.innerText = 'ERR!';

            // Ajuste para não spawnar fora da largura da tela (cite: 11125.jpg)
            const maxX = box.clientWidth - 60;
            const maxY = box.clientHeight - 60;
            
            t.style.left = Math.random() * maxX + 'px';
            t.style.top = Math.random() * maxY + 'px';

            // Suporte híbrido: Toque (mobile) e Clique (PC)
            const removeThreat = (e) => {
                e.preventDefault();
                t.style.background = "#fff"; // Feedback de acerto
                setTimeout(() => t.remove(), 50);
            };

            t.addEventListener('touchstart', removeThreat, {passive: false});
            t.addEventListener('mousedown', removeThreat);

            box.appendChild(t);

            // O vírus some se não for deletado rápido
            setTimeout(() => { if(t.parentNode) t.remove(); }, 1200);
        }

        // Começa o ataque
        setInterval(spawn, 700);
    </script>

    <script>
    (function() {
        const isMobile = /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent);
        if (isMobile) {
            let viewport = document.querySelector('meta[name="viewport"]');
            if (!viewport) {
                viewport = document.createElement('meta');
                viewport.name = "viewport";
                document.getElementsByTagName('head')[0].appendChild(viewport);
            }
            viewport.content = "width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no";
            
            document.addEventListener('touchstart', function requestFullScreen() {
                const docEl = document.documentElement;
                if(docEl.requestFullscreen) docEl.requestFullscreen();
                else if(docEl.webkitRequestFullScreen) docEl.webkitRequestFullScreen();
                document.removeEventListener('touchstart', requestFullScreen);
            }, {once: true});
        }
    })();
    </script>
